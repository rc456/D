]);

// Step 4: Apply below queries

// A) Count Query: Count all documents
db.employees.find().count();

// Count documents matching a criteria
db.employees.find({ "Dept": "IT" }).count();

// B) Sort Query: Sort by Salary in Ascending Order (1)
db.employees.find().sort({ "Salary": 1 });

// Sort by Salary in Descending Order (-1)
db.employees.find().sort({ "Salary": -1 });

// C) Limit Query: Limit to 3 documents
db.employees.find().limit(3);

// D) Skip Query: Skip the first 3 documents
db.employees.find().skip(3);

// Limit and Skip in one query: Skip 3 documents and then limit to the next 2
db.employees.find().skip(3).limit(2);

// E) Aggregate Function

// Step 4 (Match): Find documents where Dept is 'IT'
db.employees.aggregate([ { $match: { "Dept": "IT" } } ]);

// Step 5 (Group): Group by Dept
db.employees.aggregate([ { $group: { _id: "$Dept" } } ]);

// Group by Dept and sum the Salary (a common aggregation use)
db.employees.aggregate([ { $group: { _id: "$Dept", totalSalary: { $sum: "$Salary" } } } ]);

// Step 6 (Match and Count): Count employees where Dept is 'IT'
db.employees.aggregate([ { $match: { "Dept": "IT" } }, { $count: "it_employees_count" } ]);
