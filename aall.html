Aim: - Write a selenium script to switch between frames and multiple browser windows.

1.pip install selenium webdriver-manager

2.py -3.11 -m pip install selenium


code:
import time
from selenium import webdriver
from selenium.webdriver.common.by import By
from selenium.webdriver.support.ui import WebDriverWait
from selenium.webdriver.support import expected_conditions as EC

def selenium_switch_frames_and_windows(): 
    driver = webdriver.Chrome() 
    wait = WebDriverWait(driver, 10) 
    try: 
        driver.get("https://demoqa.com/browser-windows") 
        original_window = driver.current_window_handle 
        print("Main window:", original_window) 
        
        # Click button that opens a new tab 
        wait.until(EC.element_to_be_clickable((By.ID, "tabButton"))).click() 
        
        # Wait for second window 
        wait.until(EC.number_of_windows_to_be(2)) 
        
        # Switch to new window 
        for handle in driver.window_handles: 
            if handle != original_window: 
                driver.switch_to.window(handle) 
                break 
        
        print("Switched to new window:", driver.current_window_handle)
        
        # Read text from new window 
        text = wait.until( 
            EC.presence_of_element_located((By.ID, "sampleHeading"))
        ).text 
        print("New window text:", text) 
        
        driver.close() 
        driver.switch_to.window(original_window) 
        print("Back to original window") 
        
        driver.get("https://demoqa.com/frames")
        print("Switching to iframe...") 
        wait.until(EC.frame_to_be_available_and_switch_to_it("frame1")) 
        
        frame_text = wait.until( 
            EC.presence_of_element_located((By.ID, "sampleHeading")) 
        ).text 
        print("Frame text:", frame_text) 
        
        driver.switch_to.default_content() 
        print("Switched back to main page") 
        
        main_page_header = WebDriverWait(driver, 10).until( 
            EC.presence_of_element_located((By.TAG_NAME, "h1"))
        ).text 
        print("Main page header:", main_page_header) 
        
        time.sleep(2) 
    except Exception as e: 
        print("An error occurred:", e) 
    finally: 
        driver.quit() 

if __name__ == "__main__": 
    selenium_switch_frames_and_windows()

from selenium.webdriver.chrome.service import Service

driver.maximize_window()

time.sleep(2)
table = driver.find_element(By.ID, "table1")
headers = table.find_elements(By.TAG_NAME, "th")
header_text = [header.text for header in headers]
print("TABLE HEADERS:")
print(header_text)
print("-" * 50)

print("TABLE DATA:")

    row_data = [cell.text for cell in cells]
    print(row_data)

from selenium import webdriver
from selenium.webdriver.common.by import By
from selenium.webdriver.chrome.service import Service
from selenium.webdriver.common.action_chains import ActionChains
from selenium.webdriver.common.keys import Keys
from selenium.webdriver.support.ui import WebDriverWait
from selenium.webdriver.support import expected_conditions as EC
from webdriver_manager.chrome import ChromeDriverManager

# ---------------- Setup WebDriver ----------------
driver = webdriver.Chrome(service=Service(ChromeDriverManager().install()))
driver.maximize_window()
wait = WebDriverWait(driver, 15)

# Create Actions object
actions = ActionChains(driver)

# ---------------- Mouse Hover Action ----------------
driver.get("https://demoqa.com/tool-tips")
hover_button = wait.until(EC.presence_of_element_located((By.ID, "toolTipButton")))
actions.move_to_element(hover_button).perform()
print("Hovered over the button")

# ---------------- Double Click Action ----------------
driver.get("https://demoqa.com/buttons")
double_click_btn = wait.until(EC.presence_of_element_located((By.ID, "doubleClickBtn")))
actions.double_click(double_click_btn).perform()
print("Double-clicked the button")

# ---------------- Right Click (Context Click) ----------------
right_click_btn = wait.until(EC.presence_of_element_located((By.ID, "rightClickBtn")))
actions.context_click(right_click_btn).perform()
print("Right-clicked the button")

# ---------------- Keyboard Actions ----------------
driver.get("https://demoqa.com/text-box")
name_input = wait.until(EC.presence_of_element_located((By.ID, "userName")))
email_input = wait.until(EC.presence_of_element_located((By.ID, "userEmail")))

# Type name using Actions
actions.click(name_input).send_keys("Vaishnavi Gharat").perform()
print("Typed name using keyboard")

# Simulate CTRL + A (select all) and CTRL + C (copy)
actions.click(name_input).key_down(Keys.CONTROL).send_keys('a').send_keys('c').key_up(Keys.CONTROL).perform()
print("Selected and copied text")

# Paste into email field using send_keys (reliable)
actions.click(email_input).send_keys("ckt1234@gmail.com").perform()
print("Typed email using keyboard actions")

# ---------------- Finish ----------------
input("Press Enter to close browser...")
driver.quit()
driver.quit()

products = driver.find_elements(
    By.CSS_SELECTOR, "div[data-component-type='s-search-result']"
)
