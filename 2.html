#Import libraries
import tensorflow as tf
from tensorflow.keras import layers, models
import numpy as np
import matplotlib.pyplot as plt
import cv2
#Load sample dataset (MNIST for demo)
(x_train, _), (x_test, _) = tf.keras.datasets.mnist.load_data()
# Normalize
x_train = x_train / 255.
x_test = x_test / 255.
# Add channel dimension
x_train = x_train.reshape((-1, 28, 28, 1))
x_test = x_test.reshape((-1, 28, 28, 1))

#Build Autoencoder Model
input_img = layers.Input(shape=(28, 28, 1))
# Encoder
x = layers.Conv2D(32, (3,3), activation='relu', padding='same')(input_img)
x = layers.MaxPooling2D((2,2), padding='same')(x)
x = layers.Conv2D(16, (3,3), activation='relu', padding='same')(x)
encoded = layers.MaxPooling2D((2,2), padding='same')(x)
# Decoder
x = layers.Conv2D(16, (3,3), activation='relu', padding='same')(encoded)
x = layers.UpSampling2D((2,2))(x)
x = layers.Conv2D(32, (3,3), activation='relu', padding='same')(x)
x = layers.UpSampling2D((2,2))(x)
decoded = layers.Conv2D(1, (3,3), activation='sigmoid', padding='same')(x)
autoencoder = models.Model(input_img, decoded)
autoencoder.compile(optimizer='adam', loss='binary_crossentropy')
autoencoder.summary()

#Train the Autoencoder
history = autoencoder.fit(
x_train, x_train,
epochs=10,
batch_size=128,
shuffle=True,
validation_data=(x_test, x_test)
)

#Reconstruct Test Image
sample = x_test[0]
sample_input = sample.reshape(1, 28, 28, 1)
reconstructed = autoencoder.predict(sample_input)
#Show Original vs. Reconstructed
plt.figure(figsize=(6,3))
plt.subplot(1,2,1)
plt.title("Original")
plt.imshow(sample.reshape(28,28), cmap='gray')
plt.axis('off')
plt.subplot(1,2,2)
plt.title("Reconstructed")
plt.imshow(reconstructed.reshape(28,28), cmap='gray')
plt.axis('off')
plt.show()

#Compute Difference (Object Detection Map)
diff = np.abs(sample - reconstructed.reshape(28,28,1))
plt.imshow(diff.reshape(28,28), cmap='hot')
plt.title("Difference Heatmap (Object Detected!)")
plt.axis('off')
plt.show()

#Thresholding to Identify Object
threshold = 0.3
mask = diff > threshold
plt.figure(figsize=(5,5))
plt.imshow(mask.reshape(28,28), cmap='gray')
plt.title("Detected Object Mask")
plt.axis('off')
plt.show()
