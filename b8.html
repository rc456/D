Aim:- Implement clustering algorithm using map reduce.
 
Code:-

import numpy as np
from collections import defaultdict
 
# Generate Sample Dataset
np.random.seed(42)
data = np.vstack((
    np.random.randn(100, 2) + [2, 2],
    np.random.randn(100, 2) + [8, 8]))
K = 2
max_iters = 10
epsilon = 0.001
 
# Initialize Centroids
centroids = data[np.random.choice(len(data), K, replace=False)]
 
# Distance Function
def euclidean_distance(a, b):
    return np.linalg.norm(a - b)
 
# MAP FUNCTION
def mapper(data_chunk, centroids):
    mapped = []
    for point in data_chunk:
        distances = [euclidean_distance(point, c) for c in centroids]
        nearest_centroid = np.argmin(distances)
        mapped.append((nearest_centroid, point))
    return mapped
 
# REDUCE FUNCTION
def reducer(mapped_data):
    cluster_dict = defaultdict(list)
    # Shuffle + Sort (group by key)
    for key, point in mapped_data:
        cluster_dict[key].append(point)
    new_centroids = []
    for key in sorted(cluster_dict.keys()):
        new_centroids.append(np.mean(cluster_dict[key], axis=0))
    return np.array(new_centroids)
 
# Iterative MapReduce KMeans
for iteration in range(max_iters):
    print(f"Iteration {iteration+1}")
    # MAP
    mapped = mapper(data, centroids)
    # REDUCE
    new_centroids = reducer(mapped)
    # Check convergence
    shift = np.linalg.norm(new_centroids - centroids)
    print("Centroid shift:", shift)
    if shift < epsilon:
        break
    centroids = new_centroids
print("\nFinal Centroids:")
print(centroids)
